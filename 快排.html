                function quickSort(arr){
                    if(arr.length < =1){
                        return arr
                    }
                    let left = []
                    let right = []
                    let begin = 0
                    let end = arr.length-1
                    let flagi = Math.round((begin+end)/2)
                    let flag = arr[flagi]
                    for(let i = 0;i < arr.length;i++){
                        if(arr[i]>flag){
                            right.push(arr[i])
                        }else if(arr[i] < flag){
                            left.push(arr[i])
                        }
                    }
                    return quickSort(left).concat(flag,quickSort(right))
                }
